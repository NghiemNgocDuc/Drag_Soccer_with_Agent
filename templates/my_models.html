✍️ How to write a Soccer AI

🎯 Required function

def get_ai_move(state, is_player_a):
    # state        — dict describing the full game state (see below)
    # is_player_a  — True → you control Team A (attacks RIGHT toward x=1000)
    #                 False → you control Team B (attacks LEFT toward x=0)
    return player_idx, angle_deg, power
    # player_idx   — 0, 1, or 2 (which of your 3 players kicks)
    # angle_deg    — direction in degrees (0=right, 90=down, 180=left, 270=up)
    # power        — kick strength 0.0–100.0

📋 State dict keys

state["ball"]       = {"x": float, "y": float}   # ball position
state["players_a"]  = [{"x":float,"y":float}, ...]  # Team A (3 players)
state["players_b"]  = [{"x":float,"y":float}, ...]  # Team B (3 players)
state["score_a"]    = int   # Team A goals
state["score_b"]    = int   # Team B goals
state["kick_count"] = int   # total kicks so far
state["game_over"]  = bool  # True if match ended
state["winner"]     = str or None  # "A" | "B" | None
state["period"]     = str   # "regular_first" | "regular_second" |
                            # "et_first" | "et_second" | "penalties"
state["player_count"] = int # players per side (default 3)
state["half_length"]  = int # game-minutes per half (default 45)
state["win_goal_limit"] = int  # goals to win (default 5)

📦 Available imports & builtins

Full access to random, math, copy plus common builtins (min, max, sum, abs, sorted, enumerate, zip, any, all, len, range, …). OS / network / filesystem modules are blocked.

🧠 Strategy tips

# 1. Pick the right player
#    Use the player closest to the ball for quick response.
#    Use a different player to intercept / position defensively.

# 2. Aim smartly
#    Shoot at goal when close (x > 700 for Team A, x < 300 for Team B).
#    Vary y-coordinate of target so keeper can't always save.
#    Pass to teammates when far from goal.

# 3. Use power wisely
#    80–100 for shots on goal (harder = faster = less time for keeper).
#    40–60 for passes / defensive clears.
#    Low power (10–30) for short dribbles.

# 4. Play both sides
#    Don't always use player 0 — spread kicks across all 3 players.
#    Keep at least one player back for defense.

# 5. Consider game state
#    Check state["period"] — play more aggressively in extra time.
#    Check state["score_a"] vs state["score_b"] — chase or protect lead.

🎲 Example — shoot at goal

import math

MODEL_NAME  = "Shooter"
DESCRIPTION = "Kicks toward opponent's goal"

def get_ai_move(state, is_player_a):
    bx, by = state["ball"]["x"], state["ball"]["y"]
    players = state["players_a"] if is_player_a else state["players_b"]

    # Closest player to ball
    idx = min(range(3), key=lambda i: math.hypot(
        players[i]["x"] - bx, players[i]["y"] - by
    ))

    # Aim at center of opponent goal with some randomness
    goal_x = 1000.0 if is_player_a else 0.0
    goal_y = 312.0 + (random.random() - 0.5) * 80
    angle = math.degrees(math.atan2(goal_y - by, goal_x - bx))

    return idx, angle, 85.0

🎲 Example — pass & advance

import math

MODEL_NAME  = "Passer"
DESCRIPTION = "Passes to forward players"

def get_ai_move(state, is_player_a):
    bx, by = state["ball"]["x"], state["ball"]["y"]
    players = state["players_a"] if is_player_a else state["players_b"]

    # Kick with whichever player is closest
    idx = min(range(3), key=lambda i: math.hypot(
        players[i]["x"] - bx, players[i]["y"] - by
    ))

    # Find the furthest-forward teammate to pass to
    target_x = 0 if is_player_a else 1000  # opposite direction
    target = max(players, key=lambda p: p["x"] if is_player_a else -p["x"])
    dx = target["x"] - bx
    dy = target["y"] - by
    dist = math.hypot(dx, dy)

    if dist < 150:
        # Close enough — shoot
        goal_x = 1000.0 if is_player_a else 0.0
        angle = math.degrees(math.atan2(312 - by, goal_x - bx))
        return idx, angle, 90.0
    else:
        # Pass forward
        angle = math.degrees(math.atan2(dy, dx))
        return idx, angle, 60.0
